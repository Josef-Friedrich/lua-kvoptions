\directlua{
  luakeys = require('luakeys')
}

\def\luakeysdebug#1{
  \noindent
  Input:\par

  \noindent
  #1\par

  \bigskip

  \noindent
  Parsed Lua table:\par
  \bigskip

  {
    \tt
    \parindent=0pt
    \directlua{
      local result = luakeys.parse('#1')
      tex.print(luakeys.stringify_table(result))
    }
  }
}
