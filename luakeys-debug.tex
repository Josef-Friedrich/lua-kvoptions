\directlua{
  luakeys = require('luakeys')
}

\def\luakeysdebug#1{
  {
    \tt
    \parindent=0pt
    \directlua{
      local result = luakeys.parse('#1')
      tex.print(luakeys.stringify(result, true))
      luakeys.print(result)
    }
  }
}
