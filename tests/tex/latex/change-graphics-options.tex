\section{User definied macro}

\newcommand{\mygraphics}[2][]{
  \directlua{
    local keys = luakeys.parse('#1')
    keys['scale'] = 3 * keys['scale']
    tex.print(luakeys.render(keys))
    tex.print('\\includegraphics[' .. luakeys.render(keys) .. ']{#2}')
  }
}
