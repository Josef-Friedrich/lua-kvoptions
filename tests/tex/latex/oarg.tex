\documentclass{article}
\usepackage{luakeys}

\directlua{
  lk = luakeys.new()
}

\begin{document}

\newcommand{\cmdoarg}[2][]{
  \directlua{
    kv_string = '#1'
    local result = lk.parse('#1')
  }
  oarg: \directlua{tex.print(kv_string)}
  marg: #2
}

\cmdoarg[oar]g]{marg}

\NewDocumentCommand{ \cmdoargLthree } { O{} m } {
